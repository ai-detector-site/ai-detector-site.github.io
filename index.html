<script>
let samples = [];
let selectedType = null;
let learnedAIWords = {};
let learnedHumanWords = {};

document.getElementById('textInput').addEventListener('input', e => {
document.getElementById('charCount').innerText = e.target.value.length + " Zeichen";
});

function switchTab(tab, btn) {
document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
document.querySelectorAll('.tab-panel').forEach(p => p.classList.remove('active'));
document.getElementById('tab-' + tab).classList.add('active');
btn.classList.add('active');
}

function selectType(type) {
selectedType = type;
document.getElementById('btnAI').className = 'toggle-btn' + (type === 'ai' ? ' sel-ai' : '');
document.getElementById('btnHuman').className = 'toggle-btn' + (type === 'human' ? ' sel-human' : '');
}

function tokenize(text) {
const stopwords = new Set(['dass','aber','auch','sich','sind','wird','eine','einem','einen','einer','eines','nicht','oder','noch','mehr','sehr','durch','nach','ГѓВјber','beim','diese','dieser','diesem','diesen','dieses','kann','wird','wurde','haben','werden']);
return text.toLowerCase().replace(/[^a-zГѓВ¤ГѓВ¶ГѓВјГѓВџ\s]/g, ' ').split(/\s+/).filter(w => w.length > 3 && !stopwords.has(w));
}

function rebuildLearned() {
learnedAIWords = {};
learnedHumanWords = {};
samples.forEach(s => {
const words = tokenize(s.text);
const target = s.type === 'ai' ? learnedAIWords : learnedHumanWords;
words.forEach(w => { target[w] = (target[w] || 0) + 1; });
});
const aiN = samples.filter(s => s.type === 'ai').length;
const humN = samples.filter(s => s.type === 'human').length;
document.getElementById('countAI').innerText = aiN;
document.getElementById('countHuman').innerText = humN;
document.getElementById('learnStats').innerText = aiN + ' KI-Texte, ' + humN + ' Mensch-Texte gelernt';
const topWords = Object.entries(learnedAIWords).sort((a,b) => b[1]-a[1]).slice(0,10)
.map(([w,c]) => '<span style="background:#1e293b;padding:2px 6px;border-radius:4px;color:#f87171;font-size:0.65rem;">' + w + ' (' + c + ')</span>').join(' ');
document.getElementById('learnedWords').innerHTML = topWords || 'Noch keine Daten.';
}

function renderSamples() {
const list = document.getElementById('sampleList');
const empty = document.getElementById('emptyState');
list.innerHTML = '';
if (samples.length === 0) { empty.style.display = 'block'; return; }
empty.style.display = 'none';
[...samples].reverse().forEach((s, ri) => {
const i = samples.length - 1 - ri;
list.innerHTML += '<div class="sample-item"><span class="sample-text">' + s.text.substring(0,80) + '...</span><span class="badge ' + (s.type==='ai'?'badge-ai':'badge-human') + '">' + (s.type==='ai'?'KI':'Mensch') + '</span><button class="remove-btn" onclick="removeSample('+i+')">ГўВњВ•</button></div>';
});
}

function addSample() {
const text = document.getElementById('learnInput').value.trim();
if (text.length < 50) { alert('Bitte gib einen laengeren Text ein (mind. 50 Zeichen).'); return; }
if (!selectedType) { alert('Bitte waehle aus, ob der Text von KI oder Mensch stammt.'); return; }
samples.push({ text, type: selectedType });
document.getElementById('learnInput').value = '';
selectedType = null;
document.getElementById('btnAI').className = 'toggle-btn';
document.getElementById('btnHuman').className = 'toggle-btn';
rebuildLearned();
renderSamples();
}

function removeSample(i) {
samples.splice(i, 1);
rebuildLearned();
renderSamples();
}

function clearSamples() {
if (!confirm('Alle Lernbeispiele loeschen?')) return;
samples = [];
rebuildLearned();
renderSamples();
}

function getLearningBonus(text) {
if (samples.length === 0) return { bonus: 0, label: null };
const words = tokenize(text);
let aiScore = 0, humanScore = 0;
words.forEach(w => { aiScore += learnedAIWords[w] || 0; humanScore += learnedHumanWords[w] || 0; });
const total = aiScore + humanScore;
if (total === 0) return { bonus: 0, label: null };
const aiRatio = aiScore / total;
const bonus = Math.round((aiRatio - 0.5) * 40);
return { bonus, label: 'Г°ВџВ§В  Gelerntes Muster (KI: ' + aiScore + ' vs. Mensch: ' + humanScore + ')' };
}

function hashString(str) {
let hash = 0;
for (let i = 0; i < str.length; i++) { hash = ((hash << 5) - hash) + str.charCodeAt(i); hash |= 0; }
return Math.abs(hash);
}

// Remaps a raw score (0ГўВЂВ“100) so that:
// - scores near 0 stay near 0 (clearly human)
// - scores near 50 get pushed to 55ГўВЂВ“75 (ambiguous ГўВ†В’ looks like AI)
// - scores near 100 stay near 99 (clearly AI)
function boostScore(raw) {
const norm = raw / 100;
// Sigmoid-style push: compress the bottom half, expand the top half
// Uses a piecewise curve:
// below 30 ГўВ†В’ map to 5ГўВЂВ“45 (human-leaning texts stay low but not too low)
// 30ГўВЂВ“60 ГўВ†В’ map to 55ГўВЂВ“75 (middle ground pushed above 50)
// above 60 ГўВ†В’ map to 75ГўВЂВ“99 (already high stays high)
let out;
if (raw < 30) {
out = 5 + (raw / 30) * 40;
} else if (raw < 60) {
const t = (raw - 30) / 30;
out = 55 + t * 20;
} else {
const t = (raw - 60) / 40;
out = 75 + t * 24;
}
return Math.min(Math.round(out), 99);
}

function analyzeText() {
const text = document.getElementById('textInput').value.trim();
if (text.length < 50) { alert('Bitte gib einen laengeren Text ein (mind. 50 Zeichen).'); return; }

let score = 0;
let findings = [];

const buzzwords = [
{ regex: /\b(testament|delve|intricate|pivotal|underscore|landscape|foster|enhance|crucial|shaping|unlock|tapestry|vibrant|comprehensive|meticulous|realm|embark|unleash)\b/gi, label: "KI-Vokabular (z.B. 'delve', 'tapestry')", weight: 8 },
{ regex: /\b(zudem|des Weiteren|darГѓВјber hinaus|in diesem Zusammenhang|zusammenfassend lГѓВ¤sst sich sagen)\b/gi, label: "Formale Uebergaenge", weight: 5 },
{ regex: /\b(entscheidende Rolle|wesentlicher Aspekt|es ist wichtig zu beachten|nicht nur ... sondern auch)\b/gi, label: "Hohle Phrasen", weight: 6 }
];
buzzwords.forEach(item => {
const matches = text.match(item.regex);
if (matches) { score += matches.length * item.weight; findings.push({label: item.label}); }
});

const paragraphs = text.split(/\n\s*\n/).filter(p => p.trim().length > 0);
if (paragraphs.length >= 3) { score += 12; findings.push({label: 'Saubere Block-Struktur (' + paragraphs.length + ' Abschnitte)'}); }

const lastParagraph = paragraphs[paragraphs.length - 1].trim();
if (lastParagraph.endsWith('?') || /viel erfolg|zusammenfassend|hoffe|hilfreich/i.test(lastParagraph)) {
score += 18; findings.push({label: "KI-typischer Abschluss (Frage/CTA)"});
}

const bulletPoints = (text.match(/^[ГўВЂВў\-\*\d\.]\s+/gm) || []).length;
if (bulletPoints > 2) { score += bulletPoints * 4; findings.push({label: "Hohe Listen-Frequenz"}); }

if (/^(Sicher|Gerne|Hier ist|In diesem|Als|NatГѓВјrlich|Sowohl)/i.test(text.substring(0, 40))) {
score += 25; findings.push({label: "KI-Antwort-Schema (Einleitung)"});
}

const sentences = text.split(/[.!?]/).filter(s => s.trim().length > 0);
if (sentences.length > 3) {
const lens = sentences.map(s => s.trim().split(/\s+/).length);
const avg = lens.reduce((a,b) => a+b, 0) / lens.length;
const variance = lens.reduce((a,b) => a + Math.pow(b-avg,2), 0) / lens.length;
if (variance < 12) { score += 20; findings.push({label: "Monotoner Satzrhythmus"}); }
}

score += hashString(text) % 5;

const { bonus, label } = getLearningBonus(text);
if (label) { score += bonus; findings.push({label, learning: true}); }

const rawScore = Math.min(Math.round(score), 100);
const finalScore = boostScore(rawScore);

document.getElementById('resultArea').style.display = 'block';
document.getElementById('percentage').innerText = finalScore + '%';
setTimeout(() => { document.getElementById('progressBar').style.width = finalScore + '%'; }, 100);

const list = document.getElementById('markerList');
list.innerHTML = '';
findings.forEach(f => {
list.innerHTML += '<li class="sample-item"><span style="color:' + (f.learning ? '#22d3ee' : '#cbd5e1') + ';font-size:0.8rem;">' + f.label + '</span><span class="badge badge-detected">DETEKTIERT</span></li>';
});
if (findings.length === 0) {
list.innerHTML = '<li style="text-align:center;color:#475569;font-size:0.8rem;padding:1rem;">Keine Marker gefunden</li>';
}

const color = finalScore > 70 ? '#f87171' : (finalScore > 50 ? '#fbbf24' : '#34d399');
document.getElementById('percentage').style.color = color;
document.getElementById('progressBar').style.backgroundColor = color;
document.getElementById('accentBar').style.backgroundColor = color;
}
